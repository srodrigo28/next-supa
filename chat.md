#### Modelo de tabela usada
```
create table public.perfil (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  name text null,
  telefone text null,
  email text null,
  senha text null,
  papel text null default 'default'::text,
  user_id uuid null,
  avatar_url text null default 'https://encrypted-tbn0.gstatic.com/images?q=tbn:ANd9GcSvxAG7fUi955Uf_FdmG6mB7LmKzF72s9LeNA&s'::text,
  last_access_at timestamp with time zone null,
  constraint perfil_pkey primary key (id),
  constraint perfil_email_key unique (email),
  constraint perfil_nome_key unique (name),
  constraint perfil_user_ref_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;
```
